export const PyType = ({ type }) => {
const renderField = (name, details) => (

<div className="py-2 border-b border-gray-200 last:border-0">
  <div className="flex items-start justify-between">
    <div>
      <span className="font-mono text-sm text-blue-600">{name}</span>
      <span className="ml-2 font-mono text-sm text-gray-600">
        {details.type}
      </span>
    </div>
    {details.required && (
      <span className="text-xs text-red-500 font-semibold">required</span>
    )}
  </div>
  {details.description && (
    <p className="mt-1 text-sm text-gray-600">{details.description}</p>
  )}
</div>
);

return (

<div className="bg-white rounded-lg border border-gray-200 p-4 my-4">
  <div className="font-medium text-lg mb-2">{type.title}</div>
  <div className="space-y-1">
    {Object.entries(type.fields).map(([name, details]) =>
      renderField(name, details)
    )}
  </div>
</div>
); };
